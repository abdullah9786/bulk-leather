# MongoDB Atlas Connection String (Required)
# Get this from MongoDB Atlas → Database → Connect → Connect your application
MONGODB_URI=mongodb+srv://bulkleather:YOUR_PASSWORD@cluster0.xxxxx.mongodb.net/bulkleather?retryWrites=true&w=majority

# Steps to get MongoDB Atlas connection string:
# 1. Create account at https://www.mongodb.com/cloud/atlas
# 2. Create free cluster (M0)
# 3. Create database user
# 4. Whitelist IP (0.0.0.0/0 for all IPs)
# 5. Get connection string and replace YOUR_PASSWORD with your password
